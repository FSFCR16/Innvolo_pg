"use client"

import Image from "next/image"
import Link from "next/link"
import TarjetaProducto from "./TarjetaProducto"

export default function DetalleProductoCliente({ producto, relacionados, categoria, subcategoria }) {

  if (!producto) return (
    <div className="max-w-7xl mx-auto px-8 py-20 text-center">
      <p className="text-gris">Producto no encontrado.</p>
    </div>
  )

  return (
    <div className="max-w-7xl mx-auto px-8 py-12 flex flex-col gap-16">

      {/* Detalle principal */}
      <div className="grid grid-cols-2 gap-12">

        {/* Imagen */}
        <div className="relative h-[500px] overflow-hidden">
          <Image
            src={producto.imagen}
            alt={producto.nombre}
            fill
            className="object-cover"
          />
        </div>

        {/* Info */}
        <div className="flex flex-col justify-center gap-6">

          <div>
            <span className="font-cursiva text-dorado text-xl">
              {subcategoria.replace(/-/g, " ")}
            </span>
            <h1 className="font-sans font-bold text-4xl text-primary mt-1">
              {producto.nombre}
            </h1>
          </div>

          <p className="text-gris text-sm leading-relaxed">
            {producto.descripcion}
          </p>

          {/* Atributos */}
          {producto.atributos.length > 0 && (
            <div className="flex flex-col gap-3">
              {producto.atributos.map((attr) => (
                <div key={attr.nombre} className="flex gap-4 border-b border-gris/10 pb-3">
                  <span className="text-sm font-bold text-primary w-36 shrink-0">
                    {attr.nombre}
                  </span>
                  <span className="text-sm text-gris">
                    {attr.valor}
                  </span>
                </div>
              ))}
            </div>
          )}

          <Link
            href={`/contactanos?producto=${encodeURIComponent(producto.nombre)}`}
            className="mt-4 px-10 py-4 bg-primary text-white text-sm tracking-widest uppercase font-bold text-center hover:bg-dorado transition-colors duration-300 w-fit"
          >
            Solicitar cotización →
          </Link>

        </div>

      </div>

      {/* Productos relacionados */}
      {relacionados.length > 0 && (
        <div className="flex flex-col gap-6">
          <div>
            <h2 className="font-sans font-bold text-2xl text-primary">
              También te puede interesar
            </h2>
            <span className="font-cursiva text-xl text-dorado">
              productos relacionados
            </span>
          </div>
          <div className="grid grid-cols-4 gap-4">
            {relacionados.map((p) => (
              <TarjetaProducto key={p.id} producto={p} />
            ))}
          </div>
        </div>
      )}

    </div>
  )
}

"use client"

import { useState } from "react"
import TarjetaProducto from "./TarjetaProducto"
import { FaSearch } from "react-icons/fa"

export default function GridProductos({ categoria, subcategoria, productosIniciales = [] }) {
  const [busqueda, setBusqueda] = useState("")
  console.log(productosIniciales)
  const productosFiltrados = productosIniciales
    .filter((p) => !subcategoria || p.subcategoria === subcategoria)
    .filter((p) => {
      const texto = busqueda.toLowerCase()
      return (
        p.nombre.toLowerCase().includes(texto) ||
        p.descripcion?.toLowerCase().includes(texto) ||
        p.subcategoria?.toLowerCase().includes(texto)
      )
    })
  return (
    <div className="flex flex-col gap-6">

      <div className="relative">
        <FaSearch className="absolute left-3 top-1/2 -translate-y-1/2 text-gris/50" />
        <input
          type="text"
          placeholder="Buscar por nombre, descripción o subcategoría..."
          value={busqueda}
          onChange={(e) => setBusqueda(e.target.value)}
          className="w-full pl-10 pr-4 py-3 border border-gris/20 text-sm text-primary placeholder:text-gris/50 focus:outline-none focus:border-dorado transition-colors duration-200"
        />
      </div>

      <p className="text-sm text-gris">
        Mostrando {productosFiltrados.length} resultados
      </p>

      <div className="grid grid-cols-3 gap-4">
        {productosFiltrados.map((producto) => (
          <TarjetaProducto
            key={producto.id}
            producto={producto}
          />
        ))}
      </div>

    </div>
  )
}
